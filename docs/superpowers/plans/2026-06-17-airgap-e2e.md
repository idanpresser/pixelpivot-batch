def _probe_all_dimensions(self, paths: list[str]) -> Dict[str, tuple[int, int]]:
    from ..core.utils import probe_image_dimensions
    workers = min(32, (os.cpu_count() or 4) * 4)
    from concurrent.futures import ThreadPoolExecutor

    def _safe_probe(path: str) -> tuple[int, int]:
        try:
            return probe_image_dimensions(path)
        except Exception as e:
            log.warning(f"Could not probe dimensions for {Path(path).name}: {e}")
            return (0, 0)

    with ThreadPoolExecutor(max_workers=workers) as ex:
        dims = list(ex.map(_safe_probe, paths))
    return dict(zip(paths, dims))

def _win32_safe_path(path: str) -> str:
    """Prefix absolute Windows paths with \\?\ to bypass the 260-char MAX_PATH limit.

    UNC paths (\\server\share\...) become \\?\UNC\server\share\...
    Already-prefixed paths and relative paths are returned unchanged.
    On non-Windows platforms the input is returned as-is.
    """
    if sys.platform != "win32":
        return path
    import os
    if not os.path.isabs(path):
        return path
    # Normalise separators and resolve . / .. without hitting the filesystem
    from pathlib import PureWindowsPath
    s = str(PureWindowsPath(path))
    if s.startswith("\\\\?\\"):
        return s
    if s.startswith("\\\\"):
        # UNC: \\server\share\... -> \\?\UNC\server\share\...
        return "\\\\?\\UNC\\" + s[2:]
    return "\\\\?\\" + s
```

def probe_image_dimensions(path: str) -> tuple[int, int]:
    from .converters.base import _win32_safe_path
    safe = _win32_safe_path(path)
    cmd = [
        "ffprobe",
        "-v", "error",
        "-select_streams", "v:0",
        "-show_entries", "stream=width,height",
        "-of", "json",
        safe,
    ]
    try:
        creationflags = subprocess.CREATE_NO_WINDOW if sys.platform == "win32" else 0
        result = subprocess.run(
            cmd,
            capture_output=True,
            text=True,
            check=True,
            timeout=5.0,
            creationflags=creationflags,
        )
        data = json.loads(result.stdout)
        stream = data["streams"][0]
        return int(stream["width"]), int(stream["height"])
    except (subprocess.SubprocessError, FileNotFoundError, KeyError, IndexError, json.JSONDecodeError) as e:
        log.debug(f"ffprobe failed for {path} ({e}), falling back to PIL.")
        from PIL import Image
        with Image.open(safe) as img:
            return img.size
```

## Task 7: E2E Harness (`scripts/test_e2e_airgap.ps1`)

Standalone 7-phase PowerShell harness. No pytest. Requires the API to be running OR manages its own lifecycle.

**Files:**

- [ ] **Step 1: Create the script**

```powershell
# PixelPivot Air-Gap E2E Test Harness
#
# Launches the API from deploy\, runs 4 converters x AVIF over Z:\pics\real\,
# tests UNC raw path, exercises edge cases, then tears down.
#
# Exit codes:
#   0  All assertions pass
#   2  API failed to start within timeout
#   3  Matrix conversion failure (success_count < expected or failure_count > 0)
#   4  Batch did not complete within timeout
#   5  Dataset missing or incomplete
[CmdletBinding()]
param(
    [string]$DeployDir    = (Join-Path $PSScriptRoot '..\deploy'),
    [string]$PicsRoot     = 'Z:\pics',
    [string]$UncRoot      = '\\ipsds5\Share\pics',
    [string]$ApiUrl       = 'http://localhost:8000',
    [int]   $StartupSec   = 30,
    [int]   $BatchTimeout = 600,
    [int]   $ExpectedReal = 500
)
$ErrorActionPreference = 'Stop'

$report = [System.Collections.Generic.List[string]]::new()
$allPass = $true

function Step([string]$msg)   { Write-Host "`n==> $msg" -ForegroundColor Cyan }
function Pass([string]$label) { 
    $line = "PASS  $label"
    Write-Host "  [PASS] $label" -ForegroundColor Green
    $report.Add($line)
}
function Fail([string]$label, [string]$detail, [int]$code = 3) {
    $line = "FAIL  $label  ($detail)"
    Write-Host "  [FAIL] $label — $detail" -ForegroundColor Red
    $report.Add($line)
    $script:allPass = $false
}
function Abort([string]$msg, [int]$code) {
    Write-Host "ABORT: $msg" -ForegroundColor Red
    Write-Report
    exit $code
}
function Write-Report {
    $ts = Get-Date -Format "yyyy-MM-dd HH:mm:ss"
    $lines = @("PixelPivot E2E Report — $ts", ("=" * 50)) + $report + @(
        "",
        ("RESULT: " + ($script:allPass ? "PASS" : "FAIL"))
    )
    $lines | ForEach-Object { Write-Host $_ }
    $reportPath = Join-Path $DeployDir 'last_run.txt'
    $lines | Out-File $reportPath -Encoding utf8
}

$apiProc   = $null
$sharpProc = $null

function Cleanup {
    if ($apiProc)   { $apiProc   | Stop-Process -Force -ErrorAction SilentlyContinue }
    if ($sharpProc) { $sharpProc | Stop-Process -Force -ErrorAction SilentlyContinue }
    if (Test-Path (Join-Path $PicsRoot 'out')) {
        Remove-Item (Join-Path $PicsRoot 'out') -Recurse -Force -ErrorAction SilentlyContinue
    }
}

# ---- Phase 1: Preflight ----
Step "Phase 1: Preflight"
$DeployDir = Resolve-Path $DeployDir
if (-not (Test-Path (Join-Path $DeployDir 'Run-PixelPivot.ps1'))) {
    Abort "deploy\ not found at $DeployDir — run build_deploy.ps1 first" 5
}
$realCount = (Get-ChildItem (Join-Path $PicsRoot 'real') -File -ErrorAction SilentlyContinue).Count
if ($realCount -lt $ExpectedReal) {
    Abort "Z:\pics\real\ has $realCount files; need $ExpectedReal — run gen_e2e_dataset.ps1 first" 5
}
try {
    $existing = Invoke-RestMethod "$ApiUrl/" -TimeoutSec 2 -ErrorAction Stop
    Abort "Port 8000 already in use. Stop existing API before running harness." 2
} catch {}
Pass "Preflight: deploy OK, $realCount real images, port 8000 free"

# ---- Phase 2: Launch API ----
Step "Phase 2: Launch API"
$py    = Join-Path $DeployDir 'python-3.14.5-embed-amd64\python.exe'
$binPaths = @(
    (Join-Path $DeployDir 'bin\ffmpeg'),
    (Join-Path $DeployDir 'bin\magick'),
    (Join-Path $DeployDir 'bin\vips\bin'),
    (Join-Path $DeployDir 'vendor\node')
)
$env:PATH           = ($binPaths -join ';') + ';' + $env:PATH
$env:PYTHONPATH     = Join-Path $DeployDir 'vendor\site-packages'
$env:PIXELPIVOT_DB_PATH = Join-Path $DeployDir 'data\pixelpivot.db'
New-Item -ItemType Directory -Force (Join-Path $DeployDir 'data') | Out-Null

$apiArgs = "-m uvicorn app.batch_api.main:app --host 127.0.0.1 --port 8000"
$script:apiProc = Start-Process -FilePath $py -ArgumentList $apiArgs `
    -WorkingDirectory $DeployDir -PassThru -WindowStyle Hidden

$deadline = (Get-Date).AddSeconds($StartupSec)
$ready = $false
while ((Get-Date) -lt $deadline) {
    try { $r = Invoke-RestMethod "$ApiUrl/" -TimeoutSec 2 -ErrorAction Stop; $ready=$true; break } catch {}
    Start-Sleep -Seconds 2
}
if (-not $ready) { Cleanup; Abort "API did not start within ${StartupSec}s" 2 }
Pass "API started (PID $($script:apiProc.Id))"

# ---- Phase 3: Sharp Daemon ----
Step "Phase 3: Sharp Daemon"
$node  = Join-Path $DeployDir 'vendor\node\node.exe'
$sharp = Join-Path $DeployDir 'app\scripts\sharp_daemon.js'
$script:sharpProc = Start-Process -FilePath $node -ArgumentList $sharp `
    -WorkingDirectory $DeployDir -PassThru -WindowStyle Hidden
Start-Sleep -Seconds 3
if ($script:sharpProc.HasExited) {
    Fail "Sharp daemon" "exited immediately (exit $($script:sharpProc.ExitCode))"
} else {
    Pass "Sharp daemon started (PID $($script:sharpProc.Id))"
}

# ---- Helper: submit batch and poll ----
function Invoke-Batch([string]$tool, [string]$srcDir, [string]$tgtDir, [string]$format = 'avif') {
    $body = @{
        source_dir    = $srcDir
        target_dir    = $tgtDir
        target_format = $format
        tool          = $tool
        category      = 'general'
        trigger_type  = 'e2e'
    } | ConvertTo-Json
    $resp = Invoke-RestMethod "$ApiUrl/api/v1/batches" -Method Post -Body $body `
        -ContentType 'application/json' -TimeoutSec 30
    $runId = $resp.run_id
    if (-not $runId) { return $null }

    $deadline = (Get-Date).AddSeconds($BatchTimeout)
    while ((Get-Date) -lt $deadline) {
        Start-Sleep -Seconds 5
        $status = Invoke-RestMethod "$ApiUrl/api/v1/batches/$runId/status" -TimeoutSec 10
        if ($status.status -in 'completed','failed') { return $status }
    }
    return $null  # timeout
}

# ---- Phase 4: Main Matrix (4 tools x AVIF) ----
Step "Phase 4: Main matrix — 4 tools x AVIF"
$tools = @('magick','vips','sharp','ffmpeg')
foreach ($tool in $tools) {
    $tgt = Join-Path $PicsRoot "out\$tool"
    New-Item -ItemType Directory -Force $tgt | Out-Null
    $t0 = Get-Date
    $s = Invoke-Batch $tool (Join-Path $PicsRoot 'real') $tgt
    $elapsed = [int]((Get-Date) - $t0).TotalSeconds

    if (-not $s)                       { Fail "$tool AVIF mapped" "timeout after ${BatchTimeout}s" 4; continue }
    if ($s.status -eq 'failed')        { Fail "$tool AVIF mapped" "batch status=failed"; continue }
    if ($s.success_count -lt $ExpectedReal) { Fail "$tool AVIF mapped" "success=$($s.success_count) want $ExpectedReal"; continue }
    if ($s.failure_count -gt 0)        { Fail "$tool AVIF mapped" "failure_count=$($s.failure_count)"; continue }

    # Telemetry assertions
    $tel = $s.telemetry
    $cpuOk  = $tel -and $tel.cpu_avg -ne $null -and $tel.cpu_avg -ge 0
    $gpuAbs = -not ($tel | Get-Member -Name 'gpu_*' -ErrorAction SilentlyContinue)
    $telStr = if ($cpuOk) { "CPU:OK" } else { "CPU:MISSING" }
    if (-not $cpuOk) { Fail "$tool AVIF telemetry" "cpu_avg missing or null" }
    else { Pass "$tool AVIF mapped | ${elapsed}s | $ExpectedReal/${ExpectedReal} | $telStr" }
}

# ---- Phase 5: UNC Raw Path ----
Step "Phase 5: UNC raw path (magick AVIF)"
$uncSrc = "$UncRoot\real"
$uncTgt = Join-Path $PicsRoot 'out\unc_raw'
New-Item -ItemType Directory -Force $uncTgt | Out-Null
$s = Invoke-Batch 'magick' $uncSrc $uncTgt
if (-not $s)                       { Fail "UNC raw magick" "timeout" 4 }
elseif ($s.success_count -lt $ExpectedReal) { Fail "UNC raw magick" "success=$($s.success_count)" }
elseif ($s.failure_count -gt 0)    { Fail "UNC raw magick" "failures=$($s.failure_count)" }
else { Pass "UNC raw path magick | $($s.success_count)/$ExpectedReal" }

# ---- Phase 6: Edge Cases ----
Step "Phase 6: Edge cases"
$ecRoot = Join-Path $PicsRoot 'edge_cases'

function Test-Edge([string]$label, [string]$srcSubdir, [string]$expectFailGt0, [string]$expectSuccessGt0) {
    $src = Join-Path $ecRoot $srcSubdir
    $tgt = Join-Path $PicsRoot "out\edge_$($srcSubdir -replace '\\','_')"
    New-Item -ItemType Directory -Force $tgt | Out-Null
    $s = Invoke-Batch 'magick' $src $tgt
    if (-not $s) { Fail $label "timeout"; return }
    if ($expectFailGt0 -and $s.failure_count -eq 0) {
        Fail $label "expected failure_count>0, got 0"
    } elseif ($expectSuccessGt0 -and $s.success_count -eq 0) {
        Fail $label "expected success_count>0, got 0"
    } else {
        # Verify API still alive after each edge case
        try { Invoke-RestMethod "$ApiUrl/" -TimeoutSec 5 | Out-Null }
        catch { Fail $label "API unresponsive after edge case"; return }
        Pass "$label | success=$($s.success_count) fail=$($s.failure_count)"
    }
}

Test-Edge "truncated files"    "truncated"        $true  $false
Test-Edge "empty file"         "empty"            $true  $false
Test-Edge "bad header"         "bad_header"       $true  $false
Test-Edge "tiny 1x1"          "tiny"             $false $true

# Huge image — expect rejected (failure_count > 0 or success_count == 0 with specific error)
$hugeS = Invoke-Batch 'magick' (Join-Path $ecRoot 'huge') (Join-Path $PicsRoot 'out\edge_huge')
if ($hugeS -and $hugeS.success_count -eq 0) {
    Pass "huge image rejected | errors=$($hugeS.failure_count)"
} else {
    Fail "huge image rejection" "success_count=$($hugeS.success_count) — MASSIVE_IMAGE filter did not fire"
}

# Path cases: all should succeed
$pathCases = @('paths\unicode','paths\spaces','paths\deep\a\b\c\d','paths\longname')
foreach ($pc in $pathCases) {
    $src = Join-Path $ecRoot $pc
    if (Test-Path $src) {
        $s = Invoke-Batch 'magick' $src (Join-Path $PicsRoot "out\edge_path_$($pc -replace '[\\:]','_')")
        if ($s -and $s.failure_count -eq 0) { Pass "path case: $pc" }
        else { Fail "path case: $pc" "failure_count=$($s.failure_count)" }
    }
}

# ---- Phase 7: Teardown ----
Step "Phase 7: Teardown"
Cleanup
Pass "Teardown complete"

Write-Report
if (-not $allPass) { exit 3 }
exit 0
```

```powershell
# Assemble the minimum deploy\ folder for air-gapped Windows deployment.
#
# Run from the project root on an ONLINE machine that has all vendored
# binaries and the embedded Python already present.
#
# Result: deploy\ is self-contained — copy to any Windows 10+ machine,
# double-click start.bat, done.
[CmdletBinding()]
param(
    [string]$ProjectRoot = (Resolve-Path '.').Path,
    [string]$DeployDir   = 'deploy',
    [switch]$SkipManifest
)
$ErrorActionPreference = 'Stop'

function Step([string]$msg) { Write-Host "`n==> $msg" -ForegroundColor Cyan }
function Require([string]$path, [string]$label) {
    if (-not (Test-Path $path)) { throw "Missing $label at $path" }
    Write-Host "  OK: $label"
}

$DeployDir = Join-Path $ProjectRoot $DeployDir

# ---- 1. Preflight ----
Step "Preflight"
$pyEmbed = Join-Path $ProjectRoot 'python-3.14.5-embed-amd64'
Require "$pyEmbed\python.exe"          "Embedded Python"
Require (Join-Path $ProjectRoot 'vendor\wheels')     "Wheel mirror"
Require (Join-Path $ProjectRoot 'vendor\node\node.exe') "Node.js"
Require (Join-Path $ProjectRoot 'bin\ffmpeg\ffmpeg.exe') "ffmpeg"
Require (Join-Path $ProjectRoot 'bin\magick\magick.exe') "magick"
Require (Join-Path $ProjectRoot 'bin\vips\bin\vips.exe') "vips"

$wheelCount = (Get-ChildItem (Join-Path $ProjectRoot 'vendor\wheels') -Filter '*.whl').Count
if ($wheelCount -lt 10) { throw "Wheel mirror has only $wheelCount wheels — run download_wheels.ps1 first" }
Write-Host "  OK: $wheelCount wheels"

# ---- 2. Clean (preserve data\) ----
Step "Clean deploy\"
if (Test-Path $DeployDir) {
    Get-ChildItem $DeployDir -Exclude 'data' | Remove-Item -Recurse -Force
}
New-Item -ItemType Directory -Force $DeployDir | Out-Null
New-Item -ItemType Directory -Force (Join-Path $DeployDir 'data') | Out-Null

# ---- 3. Stage app source (no tests, docs, scripts) ----
Step "Stage app\"
$appSrc = Join-Path $ProjectRoot 'app'
$appDst = Join-Path $DeployDir   'app'
$robocopyExcludes = @('__pycache__', 'tests', '*.pyc')
& robocopy $appSrc $appDst /E /XD $robocopyExcludes /NFL /NDL /NJH /NJS | Out-Null

# ---- 4. Stage runtime ----
Step "Stage embedded Python"
& robocopy $pyEmbed (Join-Path $DeployDir 'python-3.14.5-embed-amd64') /E /NFL /NDL /NJH /NJS | Out-Null

Step "Stage Node.js"
$nodeSrc = Join-Path $ProjectRoot 'vendor\node'
& robocopy $nodeSrc (Join-Path $DeployDir 'vendor\node') /E /NFL /NDL /NJH /NJS | Out-Null

Step "Stage vendored binaries"
& robocopy (Join-Path $ProjectRoot 'bin') (Join-Path $DeployDir 'bin') /E /NFL /NDL /NJH /NJS | Out-Null

Step "Copy wheel mirror"
& robocopy (Join-Path $ProjectRoot 'vendor\wheels') (Join-Path $DeployDir 'vendor\wheels') /E /NFL /NDL /NJH /NJS | Out-Null

# ---- 5. Pre-install wheels ----
Step "Pre-install wheels into vendor\site-packages\"
$sitePackages = Join-Path $DeployDir 'vendor\site-packages'
New-Item -ItemType Directory -Force $sitePackages | Out-Null

$py       = Join-Path $DeployDir 'python-3.14.5-embed-amd64\python.exe'
$wheels   = Join-Path $DeployDir 'vendor\wheels'
$pyproject = Join-Path $ProjectRoot 'pyproject.toml'

& $py -m pip install `
    --no-index `
    --find-links $wheels `
    --target $sitePackages `
    --no-deps `
    (Join-Path $ProjectRoot '.') `
    2>&1 | Write-Host
if ($LASTEXITCODE -ne 0) { throw "pip install failed (exit $LASTEXITCODE)" }

# ---- 6. Write .pth so embedded Python discovers site-packages ----
Step "Write pixelpivot.pth"
# Path is relative to the embedded Python root directory
$pthContent = "..\vendor\site-packages"
$pthPath    = Join-Path $DeployDir 'python-3.14.5-embed-amd64\pixelpivot.pth'
$pthContent | Out-File $pthPath -Encoding ascii -NoNewline

# Also uncomment 'import site' in python314._pth if present (embedded Python ships with it commented)
$embedPth = Join-Path $DeployDir 'python-3.14.5-embed-amd64\python314._pth'
if (Test-Path $embedPth) {
    (Get-Content $embedPth) -replace '^#import site', 'import site' | Set-Content $embedPth
    Write-Host "  Enabled 'import site' in python314._pth"
}

# ---- 7. Copy launchers ----
Step "Copy launchers"
Copy-Item (Join-Path $ProjectRoot 'Run-PixelPivot.ps1') $DeployDir
Copy-Item (Join-Path $ProjectRoot 'start.bat')          $DeployDir
Copy-Item (Join-Path $ProjectRoot 'pyproject.toml')     $DeployDir

# ---- 8. Manifest ----
if (-not $SkipManifest) {
    Step "Generate MANIFEST.sha256"
    & (Join-Path $ProjectRoot 'scripts\manifest.ps1') -Mode create -Root $DeployDir
}

$fileCount = (Get-ChildItem $DeployDir -Recurse -File).Count
Write-Host "`nDeploy folder ready: $DeployDir ($fileCount files)" -ForegroundColor Green
```
